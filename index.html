Intégration en 2 minutes
WebSocket Euler Stream — aucune librairie requise côté browser.
// 1. Connexion WebSocket via Euler Stream
const apiKey = "votre_clé_euler";
const uniqueId = "@votre_username";

const ws = new WebSocket(
  `wss://ws.eulerstream.com?uniqueId=${uniqueId}&apiKey=${apiKey}`
);

// 2. Écouter les événements live
ws.onmessage = (event) => {
  const data = JSON.parse(event.data);
  data.messages?.forEach((msg) => {
    switch(msg.type) {
      case 'WebcastChatMessage': // Commentaire
      case 'WebcastGiftMessage': // Cadeau reçu
      case 'WebcastLikeMessage': // Like
      case 'WebcastMemberMessage': // Quelqu'un rejoint
    }
  });
};

// 3. OAuth Flow (pour apps client-facing)
const authUrl = new URL('https://www.eulerstream.com/tiktok/oauth/authorize');
authUrl.searchParams.set('client_id', 'YOUR_CLIENT_ID');
authUrl.searchParams.set('scope', 'webcast:fetch webcast:live_check');
window.location.href = authUrl.toString();
